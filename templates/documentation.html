Qrious API Documentation

Generate QR codes, WiFi QR, and barcodes with branding and logo support.

Base URL

https://qr.soltryxsolutions.com

Authentication

Public access available. API key support on request.

POST /api/generate-qr

Generate QR codes with optional logo branding and high error correction.

Request

Content-Type: application/x-www-form-urlencoded

Parameter Type Required Description
data string Yes QR content (URL, WiFi, vCard, text)
logo string (base64 / URL) No Logo source:
  • Base64 image
  • Image URL
  • File upload (multipart)
  • Fallback: static/logo.png
 
curl -X POST https://qr.soltryxsolutions.com/api/generate-qr \
-d "data=https://site.com" --output qr.png

Language Examples

Node.js (Axios)

const axios = require("axios");
const fs = require("fs");

axios.post(
  "https://qr.soltryxsolutions.com/api/generate-qr",
  new URLSearchParams({ data: "https://site.com" }),
  { responseType: "arraybuffer" }
)
.then(res => fs.writeFileSync("qr.png", res.data));
                
Python (Requests)

import requests

response = requests.post(
  "https://qr.soltryxsolutions.com/api/generate-qr",
  data={"data": "https://site.com"}
)

with open("qr.png", "wb") as f:
    f.write(response.content)
                
PHP (cURL)

$ch = curl_init("https://qr.soltryxsolutions.com/api/generate-qr");
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, "data=https://site.com");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$result = curl_exec($ch);
file_put_contents("qr.png", $result);
curl_close($ch);
                
Java (HttpClient)

HttpClient client = HttpClient.newHttpClient();

HttpRequest request = HttpRequest.newBuilder()
  .uri(URI.create("https://qr.soltryxsolutions.com/api/generate-qr"))
  .POST(HttpRequest.BodyPublishers.ofString("data=https://site.com"))
  .header("Content-Type", "application/x-www-form-urlencoded")
  .build();

HttpResponse response =
  client.send(request, HttpResponse.BodyHandlers.ofByteArray());

Files.write(Paths.get("qr.png"), response.body());

POST /api/generate-wifi-qr

Generate WiFi QR codes that auto-connect on scan with optional branding.

Parameter Type Required Description
ssid string Yes WiFi network name
password string Yes WiFi password
security string No WPA or nopass (default: WPA)
logo file / base64 / URL No Optional branding logo (white square supported) 
curl -X POST https://qr.soltryxsolutions.com/api/generate-wifi-qr \
-F "ssid=MyWiFi" \
-F "password=Secret123" \
-F "security=WPA" \
-F "file=@logo.png" \
--output wifi.png

Language Examples

Node.js

axios.post(
"https://qr.soltryxsolutions.com/api/generate-wifi-qr",
new URLSearchParams({
  ssid: "MyWiFi",
  password: "Secret123",
  security: "WPA"
}),
{ responseType: "arraybuffer" }
).then(res => fs.writeFileSync("wifi.png", res.data));
                
Python

response = requests.post(
  "https://qr.soltryxsolutions.com/api/generate-wifi-qr",
  data={
    "ssid": "MyWiFi",
    "password": "Secret123",
    "security": "WPA"
  }
)

with open("wifi.png", "wb") as f:
    f.write(response.content)
                
PHP

$fields = http_build_query([
  "ssid" => "MyWiFi",
  "password" => "Secret123",
  "security" => "WPA"
]);

$ch = curl_init("https://qr.soltryxsolutions.com/api/generate-wifi-qr");
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, $fields);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

file_put_contents("wifi.png", curl_exec($ch));
curl_close($ch);
                
Java

String form = "ssid=MyWiFi&password=Secret123&security=WPA";

HttpRequest request = HttpRequest.newBuilder()
  .uri(URI.create("https://qr.soltryxsolutions.com/api/generate-wifi-qr"))
  .POST(HttpRequest.BodyPublishers.ofString(form))
  .header("Content-Type", "application/x-www-form-urlencoded")
  .build();
